\begin{equation}
	a>1 \quad a^n>1 \quad \forall n \in \Q, n>0 \qquad
	p,q \in \N\ \land\ q \neq 0\ : n=\frac{p}{q}
\end{equation}

\begin{equation}
	a^n = \sqrt[q]{a^p} \geq \sqrt[q]{a} > \sqrt[q]{1} = 1
\end{equation}
